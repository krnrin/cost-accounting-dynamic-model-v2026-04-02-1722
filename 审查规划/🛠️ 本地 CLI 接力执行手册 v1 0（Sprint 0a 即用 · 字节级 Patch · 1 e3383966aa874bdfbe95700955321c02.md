### F0a-3 · PR-007 · sync 路由 requireRole 中间件 [security/P0]

- **目标**：任意 token 不能往服务端推/拉所有项目数据；写需要 ENGINEER+，读需要任意已登录。
- **文件**：新建 `server/src/middleware/requireRole.ts`，修改 `server/src/routes/sync.ts`。
- **Patch 1（新建 middleware）**：

```tsx
// server/src/middleware/requireRole.ts
import { Request, Response, NextFunction } from 'express';

const RANK = { VIEWER: 0, ENGINEER: 1, MANAGER: 2, ADMIN: 3 } as const;
type Role = keyof typeof RANK;

export const requireRole = (minRole: Role) => {
  return (req: Request, res: Response, next: NextFunction) => {
    const role = req.user?.role as Role | undefined;
    if (!role || !(role in RANK)) {
      res.status(403).json({ error: 'Forbidden: unknown role' });
      return;
    }
    if (RANK[role] < RANK[minRole]) {
      res.status(403).json({ error: `Forbidden: requires ${minRole}+` });
      return;
    }
    next();
  };
};
```

```tsx
function buildScenarioCompareInput(
  payload: E281ScenarioPayload,
  project: ProjectHarnessResult,
): ScenarioCompareInput {
  const costBreakdown = buildCostBreakdown(project);

  // 读 payload 已含的 quotation 期望值（seeds 内存在），否则从配置算
  const expected = payload.expectedProjectResults;
  const cost = project.vehicleCost;
  // 售价：优先 quotation.sellingPrice → expected.sellingPricePerSet → 兜底 cost*(1+利润率)
  const profitRate = Number(payload.scenario.config.costRates?.profitRate ?? 0);
  const sellingPricePerSet = Number(
    (payload as any).quotation?.sellingPricePerSet
      ?? expected?.sellingPricePerSet
      ?? cost * (1 + profitRate),
  );
  // 毛利率
  const marginRate = sellingPricePerSet > 0
    ? (sellingPricePerSet - cost) / sellingPricePerSet
    : 0;
  // 全生命周期利润 = (售价 - 成本) × 总年量
  const totalVolume = payload.project.config.volumes.reduce((s, v) => s + v.volume, 0);
  const lifecycleProfit = (sellingPricePerSet - cost) * totalVolume;

  return {
    scenarioId: payload.mode,
    scenarioName: payload.scenario.scenarioName,
    status: payload.scenario.scenarioType,
    kpis: {
      totalCostPerSet: cost,
      sellingPricePerSet,
      marginRate,
      lifecycleProfit,
      vehicleCostPerSet: cost,
      totalHarnesses: project.harnessCount,
    },
    costBreakdown,
  };
}
```

### F0a-6 · PR-039 · metal_api 铝（XAL）单位修复 30000× 误差 [engine/P0]

- **目标**：metals-api 的 XAL 是 1 USD = X **metric ton** of aluminum，不是 troy oz；当前按 troy oz × 32.15 算导致铝价×31~32k。
- **文件**：`app/src/engine/metal_api.ts`（已读 → SHA `1fbc892`）`parseMetalsApiResponse` 第 86-99 行。
- **Patch**：

```tsx
export function parseMetalsApiResponse(
  response: MetalsApiResponse,
  usdToCny: number = 7.25
): MetalPriceData | null {
  if (!response.success || !response.rates) return null;

  const cuRatePerOz = response.rates['XCU'];   // troy oz quoted
  const alRatePerTon = response.rates['XAL'];  // metric ton quoted (corrected)

  if (!cuRatePerOz || !alRatePerTon) return null;

  // 铜：1 troy oz = 31.1035 g → 1 kg = 32.1507 troy oz
  const cuUsdPerOz = 1 / cuRatePerOz;
  const cuCnyPerKg = cuUsdPerOz * 32.1507 * usdToCny;

  // 铝：metric ton = 1000 kg；不再 ×32.1507
  const alUsdPerTon = 1 / alRatePerTon;
  const alCnyPerKg = (alUsdPerTon / 1000) * usdToCny;

  return {
    copper: Math.round(cuCnyPerKg * 100) / 100,
    aluminum: Math.round(alCnyPerKg * 100) / 100,
    source: 'metals-api',
    fetchedAt: new Date().toISOString(),
    dataDate: response.date,
    fromCache: false,
  };
}
```
